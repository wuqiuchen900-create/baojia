        public Action<string> Log { get; set; }

        private void Debug(string msg) { Log?.Invoke(msg); }

        public void Export(List<Room> rooms, string projectName, string templatePath, string outputPath, QuoteConfig config)
        {
            Debug($"模板路径: {templatePath}");
            Debug($"输出路径: {outputPath}");
            Debug($"识别的房间数: {rooms.Count}");

            File.Copy(templatePath, outputPath, overwrite: true);
            Debug("模板已复制");

            using (var package = new ExcelPackage(new FileInfo(outputPath)))
            {
                var ws = package.Workbook.Worksheets.First();
                Debug($"工作表: {ws.Name}, 行数: {ws.Dimension?.Rows ?? 0}");

                int nameRow = FindRowWithKeyword(ws, 1, "工程名称");
                SetCellValue(ws, nameRow, 1, $"工程名称：{projectName}");

                var templateGroups = ParseTemplate(ws, config);
                Debug($"模板分组数: {templateGroups.Count}");

                ProcessRooms(ws, rooms, templateGroups, config);

                FixTotalFormula(ws, templateGroups);

                Debug("数据已填入，公式自动计算");

                package.Save();
                Debug("Excel 已保存");
            }
        }

        private List<TemplateGroup> ParseTemplate(ExcelWorksheet ws, QuoteConfig config)
        {
            var groups = new List<TemplateGroup>();
            TemplateGroup currentGroup = null;
            int maxRow = ws.Dimension?.Rows ?? 200;

            for (int row = 8; row <= maxRow; row++)
            {
                var col1 = (ws.Cells[row, 1].Value?.ToString() ?? "").Trim();
                var col2 = (ws.Cells[row, 2].Value?.ToString() ?? "").Trim();

                if (IsGroupHeader(col1))
                {
                    currentGroup = new TemplateGroup
                    {
                        HeaderRow = row,
                        Name = col2,
                        RoomType = ClassifyRoom(col2, col2, config),
                        FloorLevel = ExtractFloorLevel(col2, config),
                    };
                    groups.Add(currentGroup);
                    continue;
                }

                if (col1 == "" && col2.Contains("小计"))
                {
                    if (currentGroup != null)
                        currentGroup.SubtotalRow = row;
                    continue;
                }

                if (col1 == "" && col2.Contains("合计"))
                    break;

                if (currentGroup != null && !string.IsNullOrEmpty(col1) && int.TryParse(col1, out _))
                {
                    currentGroup.Items.Add(new TemplateItem
                    {
                        Row = row,
                        Name = col2,
                        Unit = (ws.Cells[row, 4].Value?.ToString() ?? "").Trim(),
                    });
                }
            }

            return groups;
        }

        /// <summary>
        /// 从文字中提取楼层（"一楼主卧" → "一楼"），通过 config.TemplateSettings.FloorAliasMap 归一化。
        /// </summary>
        private string ExtractFloorLevel(string text, QuoteConfig config)
        {
            if (string.IsNullOrWhiteSpace(text)) return "";
            var aliasMap = config?.TemplateSettings?.FloorAliasMap;
            if (aliasMap == null || aliasMap.Count == 0) return "";
            var match = Regex.Match(text, @"(?:^\s*|[^一-龥\d])(1F|2F|3F|4F|5F|一楼|二楼|三楼|四楼|五楼|一层|二层|三层|首层)(?=\s|$|[^一-龥\d])");
            if (!match.Success) return "";
            string raw = match.Groups[1].Value;
            return aliasMap.TryGetValue(raw, out var norm) ? norm : raw;
        }

        private string ClassifyRoom(string text, string fallback, QuoteConfig config)
        {
            if (string.IsNullOrWhiteSpace(text)) return fallback;
            foreach (var map in config.RoomTypeMaps)
                foreach (var kw in map.Keywords)
                    if (text.Contains(kw)) return map.RoomType;
            return fallback;
        }

        private void ProcessRooms(ExcelWorksheet ws, List<Room> rooms,
            List<TemplateGroup> templateGroups, QuoteConfig config)
        {
            // CAD 房间按 (楼层, 类型) 复合键分组
            var cadGroups = rooms
                .GroupBy(r => (r.FloorLevel, r.RoomType))
                .ToDictionary(g => g.Key, g => g.ToList());

            // 模板分组按 (楼层, 类型) 复合键建立索引（仅有效分组）
            var tplGroups = templateGroups
                .Where(g => g.SubtotalRow > g.HeaderRow)
                .GroupBy(g => (g.FloorLevel, g.RoomType))
                .ToDictionary(g => g.Key, g => g.ToList());

            foreach (var kv in cadGroups)
            {
                var cadFloor = kv.Key.Item1;
                var cadType = kv.Key.Item2;
                var cadRooms = kv.Value;

                var tplList = ResolveTemplates(cadFloor, cadType, tplGroups, templateGroups);
                if (tplList == null || tplList.Count == 0)
                {
                    Debug($"  类型 [{cadType}] 楼层 [{cadFloor}]: 模板中无匹配分组，跳过 {cadRooms.Count} 个房间");
                    continue;
                }

                while (tplList.Count < cadRooms.Count)
                {
                    var last = tplList[tplList.Count - 1];
                    var clone = CloneGroup(ws, last);
                    if (clone == null) break;
                    templateGroups.Add(clone);
                    tplList.Add(clone);

                    int addedRows = clone.SubtotalRow - clone.HeaderRow + 1;
                    foreach (var g in templateGroups)
                    {
                        if (g.HeaderRow > last.SubtotalRow && g != clone)
                        {
                            g.HeaderRow += addedRows;
                            g.SubtotalRow += addedRows;
                            foreach (var item in g.Items)
                                item.Row += addedRows;
                        }
                    }
                    Debug($"  复制分组: [{last.Name}] -> [{clone.Name}]");
                }

                for (int i = 0; i < Math.Min(cadRooms.Count, tplList.Count); i++)
                {
                    var room = cadRooms[i];
                    var group = tplList[i];

                    ws.Cells[group.HeaderRow, 2].Value = room.Name;
                    Debug($"  房间 [{room.Name}] 楼层 [{room.FloorLevel}] -> 模板分组 [{group.Name}] -> 改名 [{room.Name}]");

                    FillRoomData(ws, group, room, config);
                }
            }
        }

        /// <summary>
        /// 三级查找：exact → 通用模板（floor=""） → 首次楼层（默认一楼）
        /// </summary>
        private List<TemplateGroup> ResolveTemplates(
            string cadFloor, string cadType,
            Dictionary<(string Floor, string Type), List<TemplateGroup>> tplGroups,
            List<TemplateGroup> allGroups)
        {
            // 1. exact
            if (!string.IsNullOrEmpty(cadFloor)
                && tplGroups.TryGetValue((cadFloor, cadType), out var exact) && exact.Count > 0)
                return exact;
            // 2. flat（CAD 写了楼层, 模板只有通用模板）
            if (!string.IsNullOrEmpty(cadFloor)
                && tplGroups.TryGetValue(("", cadType), out var flat) && flat.Count > 0)
            {
                Debug($"  CAD楼层 [{cadFloor}] 回退到通用模板: {cadType}");
                return flat;
            }
            // 3. CAD 没写楼层时，找最低楼层（"一楼" 优先）
            if (string.IsNullOrEmpty(cadFloor))
            {
                var oneF = allGroups
                    .Where(g => g.RoomType == cadType && g.SubtotalRow > g.HeaderRow && g.FloorLevel == "一楼")
                    .ToList();
                if (oneF.Count > 0)
                {
                    Debug($"  CAD无楼层，回退到模板一楼: {cadType}");
                    return oneF;
                }
                // 没有"一楼"，按 FloorLevel 汉字序取第一个
                var first = allGroups
                    .Where(g => g.RoomType == cadType && g.SubtotalRow > g.HeaderRow && !string.IsNullOrEmpty(g.FloorLevel))
                    .OrderBy(g => g.FloorLevel)
                    .ToList();
                if (first.Count > 0)
                {
                    Debug($"  CAD无楼层，回退到模板首个楼层 [{first[0].FloorLevel}]: {cadType}");
                    return first;
                }
            }
            // 4. 兜底：找任意匹配类型的模板（已在 tplGroups 中）
            return null;
        }

        private TemplateGroup CloneGroup(ExcelWorksheet ws, TemplateGroup source)
        {
            if (source.SubtotalRow <= source.HeaderRow)
            {
                Debug($"  [!] 跳过复制 [{source.Name}]: 分组无小计行");
                return null;
            }
            int totalRows = source.SubtotalRow - source.HeaderRow + 1;
            int insertAt = source.SubtotalRow + 1;

            ws.InsertRow(insertAt, totalRows);

            for (int r = 0; r < totalRows; r++)
            {
                int srcRow = source.HeaderRow + r;
                int dstRow = insertAt + r;
                for (int c = 1; c <= 9; c++)
                {
                    var srcCell = ws.Cells[srcRow, c];
                    var dstCell = ws.Cells[dstRow, c];
                    if (!string.IsNullOrEmpty(srcCell.FormulaR1C1))
                        dstCell.FormulaR1C1 = srcCell.FormulaR1C1;
                    else
                        dstCell.Value = srcCell.Value;
                    dstCell.Style.Numberformat.Format = srcCell.Style.Numberformat.Format;
                    dstCell.Style.Font.Name = srcCell.Style.Font.Name;
                    dstCell.Style.Font.Size = srcCell.Style.Font.Size;
                    dstCell.Style.Font.Bold = srcCell.Style.Font.Bold;
                    dstCell.Style.Border.Top.Style = srcCell.Style.Border.Top.Style;
                    dstCell.Style.Border.Bottom.Style = srcCell.Style.Border.Bottom.Style;
                    dstCell.Style.Border.Left.Style = srcCell.Style.Border.Left.Style;
                    dstCell.Style.Border.Right.Style = srcCell.Style.Border.Right.Style;
                    dstCell.Style.Fill.PatternType = srcCell.Style.Fill.PatternType;
                    var srcRgb = srcCell.Style.Fill.BackgroundColor.Rgb;
                    var srcIndexed = srcCell.Style.Fill.BackgroundColor.Indexed;
                    if (!string.IsNullOrEmpty(srcRgb))
                    {
                        try
                        {
                            string hex = srcRgb.Length == 8 ? srcRgb.Substring(2) : srcRgb;
                            dstCell.Style.Fill.BackgroundColor.SetColor(
                                System.Drawing.ColorTranslator.FromHtml("#" + hex));
                        }
                        catch { }
                    }
                    else if (srcIndexed > 0)
                    {
                        dstCell.Style.Fill.BackgroundColor.Indexed = srcIndexed;
                    }
                }
            }

            var clone = new TemplateGroup
            {
                HeaderRow = insertAt,
                SubtotalRow = insertAt + totalRows - 1,
                Name = source.Name,
                RoomType = source.RoomType,
                FloorLevel = source.FloorLevel,
            };

            int rowOffset = insertAt - source.HeaderRow;
            foreach (var item in source.Items)
            {
                clone.Items.Add(new TemplateItem
                {
                    Row = item.Row + rowOffset,
                    Name = item.Name,
                    Unit = item.Unit,
                });
            }

            return clone;
        }

        private void FixTotalFormula(ExcelWorksheet ws, List<TemplateGroup> groups)
        {
            int totalRow = 0;
            int maxRow = ws.Dimension?.Rows ?? 200;
            for (int row = 8; row <= maxRow; row++)
            {
                var col2 = (ws.Cells[row, 2].Value?.ToString() ?? "").Trim();
                if (col2.Contains("合计"))
                {
                    totalRow = row;
                    break;
                }
            }
            if (totalRow == 0) return;

            var subtotalRows = groups
                .Where(g => g.SubtotalRow > g.HeaderRow)
                .Select(g => g.SubtotalRow)
                .OrderBy(r => r)
                .ToList();

            var matRefs = string.Join("+", subtotalRows.Select(r => $"F{r}"));
            var laborRefs = string.Join("+", subtotalRows.Select(r => $"H{r}"));

            ws.Cells[totalRow, 6].Formula = $"={matRefs}";
            ws.Cells[totalRow, 8].Formula = $"={laborRefs}";
            Debug($"  合计公式已更新: 包含 {subtotalRows.Count} 个小计行");
        }

        private int FillRoomData(ExcelWorksheet ws, TemplateGroup group, Room room, QuoteConfig config)
        {
            int filled = 0;
            foreach (var item in group.Items)
            {
                if (IsFloorItem(item.Name, config))
                {
                    ws.Cells[item.Row, 3].Value = Math.Round(room.FloorArea, 2);
                    filled++;
                    Debug($"    填地面 行{item.Row}: [{item.Name}] 数量={room.FloorArea:F2}");
                }
                else if (IsWallItem(item.Name, config))
                {
                    ws.Cells[item.Row, 3].Value = Math.Round(room.WallArea, 2);
                    filled++;
                    Debug($"    填墙面 行{item.Row}: [{item.Name}] 数量={room.WallArea:F2}");
                }
            }
            return filled;
        }

        private List<string> GetFloorKeywords(QuoteConfig config)
        {
            var kws = config?.TemplateSettings?.FloorItemKeywords;
            if (kws != null && kws.Count > 0) return kws;
            return new List<string> { "铺地砖", "地砖", "地板", "正铺", "地面保护" };
        }

        private List<string> GetWallKeywords(QuoteConfig config)
        {
            var kws = config?.TemplateSettings?.WallItemKeywords;
            if (kws != null && kws.Count > 0) return kws;
            return new List<string> { "墙顶面基层加固", "墙面基层处理", "鸟巢腻子", "芬琳芬华", "五合一", "内墙乳胶漆" };
        }

        private bool IsFloorItem(string name, QuoteConfig config)
        {
            if (string.IsNullOrEmpty(name)) return false;
            return GetFloorKeywords(config).Any(k => name.Contains(k));
        }

        private bool IsWallItem(string name, QuoteConfig config)
        {
            if (string.IsNullOrEmpty(name)) return false;
            return GetWallKeywords(config).Any(k => name.Contains(k));
        }

        private bool IsGroupHeader(string col1)
        {
            if (string.IsNullOrEmpty(col1)) return false;
            string[] markers = { "一", "二", "三", "四", "五", "六", "七", "八", "九", "十" };
            if (markers.Contains(col1)) return true;
            string[] withComma = { "一、", "二、", "三、", "四、", "五、", "六、", "七、", "八、", "九、", "十、" };
            return withComma.Any(p => col1.StartsWith(p));
        }

        private int FindRowWithKeyword(ExcelWorksheet ws, int col, string keyword)
        {
            int maxRow = ws.Dimension?.Rows ?? 200;
            for (int row = 1; row <= maxRow; row++)
                if ((ws.Cells[row, col].Value?.ToString() ?? "").Contains(keyword))
                    return row;
            return 0;
        }

        private void SetCellValue(ExcelWorksheet ws, int row, int col, string value)
        {
            if (row > 0) ws.Cells[row, col].Value = value;
        }
    }

        /// 从 JSON 文件加载配置
        /// </summary>
        public static QuoteConfig Load(string path)
        {
            if (!File.Exists(path))
            {
                var defaultConfig = CreateDefault();
                defaultConfig.Save(path);
                return defaultConfig;
            }

            var json = File.ReadAllText(path);
            var cfg = JsonConvert.DeserializeObject<QuoteConfig>(json);
            // 向下兼容：旧配置没有 TemplateSettings 时填默认值
            if (cfg.TemplateSettings == null)
                cfg.TemplateSettings = new TemplateSettingsConfig();
            return cfg;
        }

        /// <summary>
        /// 保存为 JSON 文件
        /// </summary>
        public void Save(string path)
        {
            var json = JsonConvert.SerializeObject(this, Formatting.Indented);
            File.WriteAllText(path, json);
        }

namespace BaoJiaCAD
{
    /// <summary>
    /// 识别出的房间信息
    /// </summary>
    public class Room
    {
        /// <summary>房间名称，来自 CAD 中的文字</summary>
        public string Name { get; set; }

        /// <summary>房间类型：客厅、厨房、卧室等</summary>
        public string RoomType { get; set; }

        /// <summary>楼层（如 "一楼"/"二楼"/""），用于复式多层匹配</summary>
        public string FloorLevel { get; set; } = "";

        /// <summary>地面面积，单位 M2</summary>
        public double FloorArea { get; set; }

        /// <summary>房间边界周长，单位 M</summary>
        public double Perimeter { get; set; }

        /// <summary>墙面计算高度，单位 mm</summary>
        public double WallHeight { get; set; } = 2800.0;

        /// <summary>墙面工程量 = 地面面积 + 周长 × 高度</summary>
        public double WallArea => FloorArea + Perimeter * (WallHeight / 1000.0);

        /// <summary>该房间对应的报价项目</summary>
        public List<QuoteItem> Items { get; set; } = new List<QuoteItem>();
    }

    /// <summary>
    /// 报价项目
    /// </summary>
    public class QuoteItem
    {
        public string Name { get; set; }
        public string Unit { get; set; }
        public double Quantity { get; set; }
        public double MaterialPrice { get; set; }
        public double MaterialTotal => Quantity * MaterialPrice;
        public double LaborPrice { get; set; }
        public double LaborTotal => Quantity * LaborPrice;
        public bool IsSummaryItem { get; set; }
        public string Description { get; set; }
    }
